                procedure SEARCH(vertex v)
                    begin
                        mark v ``old''
                        for each vertex on L[v] do
                        if (w is marked ``new'') then
                            add (v, w) to T;
                            SEARCH(w)
                    end
            \end{lstlisting}

                algorithm DFS(graph G = (V, E))
                begin
                    T $\leftarrow \emptyset$
                    for all v in V do mark v ``new''
                    while (there exists v in v marked ``new'') do
                    SEARCH(v)
                end
            \end{lstlisting}
