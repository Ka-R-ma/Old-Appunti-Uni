            begin
                for k $\leftarrow$ 1 to n - 1 do 
                    m $\leftarrow$ k

                    for j $\leftarrow$ k + 1 to n do
                        if (A[j] < A[m]) then m $\leftarrow$ j
                    switch A[m] and A[k]
            end
