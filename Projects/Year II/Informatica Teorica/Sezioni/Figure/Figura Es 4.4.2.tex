        (q0) edge [bend left] node {a} (qa)

        (qc) edge [loop above, right] node {c} ()

        (qd) edge [] node {f} (qf)

        (qb) edge [bend left] node {a} (qa)
        (qb) edge [bend left] node {c} (qc)
