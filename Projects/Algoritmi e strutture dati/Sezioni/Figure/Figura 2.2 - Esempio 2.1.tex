                read r1
                if r1 $\le$ 0 then write 0;
                else
                    begin
                        r2 $\leftarrow$ r1;
                        r3 $\leftarrow$ r1 - 1;
                        while r3 > 0 do
                            begin
                                r2 $\leftarrow$ r2 * r1;
                                r3 $\leftarrow$ r3 - 1;
                            end
                        write r2;
                    end
            end
