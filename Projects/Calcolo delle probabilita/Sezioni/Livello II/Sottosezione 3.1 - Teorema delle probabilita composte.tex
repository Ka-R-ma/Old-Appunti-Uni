\begin{Theorem}
    Dati due eventi $E \text{e} H, H \neq 0$, siano $(\Prob{H}, \Prob{E}[H], \Prob{EH})$ valutazioni di probabilit√† sugli eventi $\Set{H, E \land H, (E \Given H)}$,

    $$
        \Prob{EH} = \Frac{\Prob{E}[H]}{\Prob{H}}

\end{Theorem}
\begin{Corollary}
    Se $\Prob{H} \neq 0$, allora
    $$
        \Prob{E}[H] = \Frac{\Prob{EH}}{\Prob{H}}
    $$
\end{Corollary}
